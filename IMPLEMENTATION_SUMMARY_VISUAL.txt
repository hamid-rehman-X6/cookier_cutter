Using Python + Requests:
   
   import requests
   
   # Create
   response = requests.post('http://localhost:8000/api/users/', json={
       'email': 'user@example.com',
       'name': 'John',
       'password': 'pass123'
   })
   user_id = response.json()['id']
   
   # Read
   response = requests.get(f'http://localhost:8000/api/users/{user_id}/')
   
   # Update
   response = requests.patch(f'http://localhost:8000/api/users/{user_id}/', json={
       'name': 'Jane'
   })
   
   # Delete
   response = requests.delete(f'http://localhost:8000/api/users/{user_id}/')

Using Django ORM:
   
   from cookie_cutter.users.models import User
   
   # Create
   user = User.objects.create_user(
       email='user@example.com',
       name='John',
       password='pass123'
   )
   
   # Read
   user = User.objects.get(email='user@example.com')
   
   # Update
   user.name = 'Jane'
   user.save()
   
   # Delete
   user.delete()
